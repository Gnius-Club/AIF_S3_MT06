# --- Configuraci√≥n del bus I2C y el sensor BME280 ---

# I2C(ID, SCL, SDA, FREQ) -- Los pines I2C del Micro:bit son P19 (SCL) y P20 (SDA)

# --- Funci√≥n para conectar a Wi-Fi (reutilizada de sesi√≥n anterior) ---

def connect_wifi():

    wlan = WLAN(STA_IF)

    wlan.active(True)

    if not wlan.isconnected():

        print('Intentando conectar a Wi-Fi...')

        display.scroll("WIFI...")

        wlan.connect(SSID, PASSWORD)

        timeout = 0

        while not wlan.isconnected() and timeout < 30: # Aumentado timeout para mayor robustez

            led_actividad.value(timeout % 2) # Parpadear LED rojo durante la espera

            time.sleep_ms(300)

            timeout += 1

    if wlan.isconnected():

        print('Conexi√≥n Wi-Fi exitosa:', wlan.ifconfig())

        display.scroll("CONECTADO")

        led_actividad.value(1) # Encender LED fijo

        return True

    else:

        print('Fallo la conexi√≥n Wi-Fi.')

        display.scroll("WIFI ERROR")

        led_actividad.value(0) # Apagar LED

        return False

# --- Funci√≥n para enviar m√∫ltiples datos a ThingSpeak ---

def send_multi_data_to_thingspeak(temp, hum, pres):

    try:

        # Construir la URL con m√∫ltiples campos

        # ¬°Aseg√∫rense que field1, field2, field3 est√°n habilitados en su canal de ThingSpeak!

        url_with_params = (THINGSPEAK_URL + '?api_key=' + API_KEY +

                           '&field1=' + str(temp) +

                           '&field2=' + str(hum) +

                           '&field3=' + str(pres))


        print("Enviando datos:", temp, hum, pres, "a:", url_with_params)

        display.scroll("Enviando...") # Mensaje en Micro:bit

        led_actividad.value(0) # Apagar LED temporalmente para indicar que est√° ocupado

        response = urequests.get(url_with_params)

        led_actividad.value(1) # Encender LED al terminar


        if response.status_code == 200:

            print("Datos enviados exitosamente. Respuesta:", response.text)

            display.show(Image.CHECK) # Indicador visual de √©xito

            time.sleep_ms(500)

            display.clear()

            return True

        else:

            print("Error al enviar datos. C√≥digo:", response.status_code, "Mensaje:", response.text)

            display.show(Image.NO) # Indicador visual de error

            time.sleep(1)

            display.clear()

            return False

    except Exception as e:

        print("Excepci√≥n al enviar datos:", e)

        display.show(Image.SAD)

        time.sleep(1)

        display.clear()

        return False

# --- Bucle principal del programa ---

if connect_wifi():
